pwd — показывает текущую директорию.
cd ДИРЕКТОРИЯ — перейти в указанную директорию.
cd .. — перейти на уровень выше.
cd ~ — перейти в домашнюю директорию.
cd / — перейти в корень файловой системы.
ls — показать содержимое текущей директории.
ls -l — подробный список файлов и папок.
ls -a — показать скрытые файлы.
mkdir НАЗВАНИЕ — создать новую директорию.
mkdir -p путь/к/директории — создать вложенные директории.
rmdir НАЗВАНИЕ — удалить пустую директорию.
rm -r НАЗВАНИЕ — удалить директорию с её содержимым.
mv ИСТОЧНИК НАЗНАЧЕНИЕ — переименовать или переместить директорию.
cp -r ИСТОЧНИК НАЗНАЧЕНИЕ — скопировать директорию с содержимым.
chmod 755 НАЗВАНИЕ — изменить права доступа к директории.
chown пользователь:групп а НАЗВАНИЕ — изменить владельца директории.
